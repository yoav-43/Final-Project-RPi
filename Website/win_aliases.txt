# PREREQUISITES (run once before using these aliases):
#   1. Install Heroku CLI:  winget install Heroku.HerokuCLI
#   2. Login to Heroku:     heroku login
#      (If MFA is enabled, generate an API token at dashboard.heroku.com/account/applications
#       and run: $env:HEROKU_API_KEY = "<your_token>")
#   3. Add Heroku remote:   heroku git:remote -a wakeup
#
# To load these aliases, add to your PowerShell profile ($PROFILE):
#   . C:\path\to\Final-Project-RPi\Website\win_aliases.txt

# 1. DEPLOYMENT ALIAS
# Description: Deploys only the website subdirectory to Heroku main branch.
# Usage: Type 'deploy' to update the live dashboard.
function deploy { 
    Write-Host "[DEPLOY] Pushing Website/wake_up to Heroku main branch..." -ForegroundColor Cyan
    git subtree push --prefix Website/wake_up heroku main 
}

# 2. DATABASE INITIALIZATION ALIAS
# Description: Resets the remote PostgreSQL schema on Heroku.
# Usage: Type 'db-init' to clear history and rebuild tables.
function db-init { 
    Write-Host "[DB-INIT] Resetting remote PostgreSQL schema..." -ForegroundColor Yellow
    heroku run python init_db.py 
}

# 3. GLOBAL BACKUP ALIAS
# Description: Stages all changes and pushes full project to GitHub origin/main.
# Usage: Type 'backup' to sync RPi code and Website code to GitHub.
function backup { 
    Write-Host "[BACKUP] Syncing full project to GitHub origin/main..." -ForegroundColor Green
    git add . 
    git commit -m "Full project backup and synchronization" 
    git push origin main 
}

# 4. START DRIVE ALIAS
# Description: Runs the driver monitor and saves all output to latest.log.
# Usage: Type 'start_drive' from the project root.
function start_drive {
    python3 monitor/monitor.py 2>&1 | Tee-Object -FilePath latest.log
}
